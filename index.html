const BACKEND_URL = "https://binjo-backend.onrender.com"; /* STATE */ let products = []; let cart = []; let wish = []; /* LOAD DATA FROM BACKEND */ async function loadProducts(){ try { const res = await fetch(`${BACKEND_URL}/products`); products = await res.json(); render(); } catch (err) { console.log("Backend error:", err); } } /* LOGIN (simple local) */ function login(){ let u = document.getElementById("user").value; let p = document.getElementById("pass").value; if(u && p){ localStorage.setItem("user",u); document.getElementById("loginBox").style.display="none"; } } /* RENDER PRODUCTS */ function render(){ let search = document.getElementById("search").value?.toLowerCase() || ""; let box = document.getElementById("products"); box.innerHTML=""; products .filter(p => p.name.toLowerCase().includes(search)) .forEach(p=>{ box.innerHTML += `

${p.name}

KSh ${p.price}
`; }); renderCart(); renderWish(); } /* CART */ function addCart(id){ let item = products.find(p=>p.id===id); cart.push(item); render(); } function removeItem(i){ cart.splice(i,1); render(); } /* CART UI */ function renderCart(){ let box = document.getElementById("cart"); box.innerHTML=""; let total=0; cart.forEach((c,i)=>{ total += c.price; box.innerHTML += `
${c.name} - KSh ${c.price}
`; }); document.getElementById("total").innerText = "Total: KSh " + total; document.getElementById("wa").href = `https://wa.me/254700000000?text=${encodeURIComponent( "Hello BINJO, I want to order. Total: KSh " + total )}`; } /* WISHLIST */ function addWish(id){ let item = products.find(p=>p.id===id); wish.push(item); render(); } function renderWish(){ let box = document.getElementById("wish"); box.innerHTML=""; wish.forEach(w=>{ box.innerHTML += `
${w.name}
`; }); } /* INIT */ loadProducts();
